<script>
    // --- CONFIGURATION ---
    // Change this filename if your CSV file is named differently
    const CSV_FILENAME = 'open26_1_thailand.csv';
    
    let rawData = [];
    let currentData = [];
    let charts = {};
    let sortCol = 'score';
    let sortAsc = false;

    // --- UTILITY FUNCTIONS ---
    // A simple CSV parser (handles basic commas, ignores commas inside quotes)
    function parseCSV(text) {
        const lines = text.trim().split('\n');
        const headers = lines[0].split(',').map(h => h.trim().replace(/^"|"$/g, ''));
        const result = [];
        
        for(let i = 1; i < lines.length; i++) {
            // Regex to split by comma, but keep commas inside quotes intact
            const row = lines[i].match(/(".*?"|[^",\s]+)(?=\s*,|\s*$)/g) || lines[i].split(',');
            if(row.length === 0 || !row[0]) continue;
            
            const obj = {};
            headers.forEach((header, index) => {
                let val = row[index] ? row[index].trim().replace(/^"|"$/g, '') : '';
                // Try to parse numbers where appropriate
                if (header === 'score' || header === 'rank') {
                    val = parseFloat(val) || 0;
                }
                obj[header] = val;
            });
            result.push(obj);
        }
        return result;
    }

    // Math statistics helpers
    const getMean = arr => arr.length ? arr.reduce((a, b) => a + b, 0) / arr.length : 0;
    const getMedian = arr => {
        if (!arr.length) return 0;
        const s = [...arr].sort((a, b) => a - b);
        const mid = Math.floor(s.length / 2);
        return s.length % 2 !== 0 ? s[mid] : (s[mid - 1] + s[mid]) / 2;
    };
    const getStdDev = (arr, mean) => {
        if(arr.length < 2) return 0;
        const variance = arr.reduce((acc, val) => acc + Math.pow(val - mean, 2), 0) / arr.length;
        return Math.sqrt(variance);
    };
    const getPercentile = (arr, p) => {
        if (!arr.length) return 0;
        const s = [...arr].sort((a, b) => a - b);
        const index = Math.ceil((p / 100) * s.length) - 1;
        return s[index];
    };

    // --- MAIN LOGIC ---
    async function loadData() {
        try {
            const response = await fetch(CSV_FILENAME);
            if (!response.ok) throw new Error("Could not load CSV");
            const text = await response.text();
            rawData = parseCSV(text);
            
            document.getElementById('loading').style.display = 'none';
            document.getElementById('dashboard-content').style.display = 'block';
            
            applyFilters(); // Initial render
        } catch (error) {
            document.getElementById('loading').innerHTML = `Error loading data. <br><small>If running locally, ensure you use a local web server (e.g. Live Server) due to CORS restrictions.</small>`;
            console.error(error);
        }
    }

    function applyFilters() {
        const gender = document.getElementById('filter-gender').value;
        const division = document.getElementById('filter-division').value;

        currentData = rawData.filter(row => {
            const matchGender = gender === 'all' || row.gender === gender;
            const matchDiv = division === 'all' || row.division === division;
            return matchGender && matchDiv;
        });

        updateDashboard();
    }

    function updateDashboard() {
        const scores = currentData.map(d => d.score);
        
        // 1. Update Cards
        const mean = getMean(scores);
        document.getElementById('val-total').innerText = currentData.length;
        document.getElementById('val-mean').innerText = mean.toFixed(1);
        document.getElementById('val-median').innerText = getMedian(scores).toFixed(1);
        document.getElementById('val-top').innerText = scores.length ? Math.max(...scores) : 0;
        document.getElementById('val-stddev').innerText = getStdDev(scores, mean).toFixed(1);
        document.getElementById('val-p90').innerText = getPercentile(scores, 90).toFixed(1); // 90th percentile (Top 10% cutoff if higher is better)

        // 2. Update Charts
        updateCharts();

        // 3. Update Table
        renderTable();
    }

    function updateCharts() {
        Chart.defaults.font.family = 'system-ui, -apple-system, sans-serif';
        Chart.defaults.color = '#6b7280';

        // Histogram Data (10 bins)
        const scores = currentData.map(d => d.score);
        const min = Math.min(...scores) || 0;
        const max = Math.max(...scores) || 1;
        const binSize = (max - min) / 10 || 1;
        const bins = Array(10).fill(0);
        const labels = Array(10).fill('').map((_, i) => `${Math.floor(min + i * binSize)} - ${Math.floor(min + (i + 1) * binSize)}`);
        
        scores.forEach(s => {
            let index = Math.floor((s - min) / binSize);
            if (index >= 10) index = 9; // Catch max value
            bins[index]++;
        });

        renderChart('histogramChart', 'bar', labels, bins, 'Score Distribution', '#3b82f6');

        // Gender Avg
        const males = currentData.filter(d => d.gender === 'M').map(d => d.score);
        const females = currentData.filter(d => d.gender === 'F').map(d => d.score);
        renderChart('genderChart', 'bar', ['Male', 'Female'], [getMean(males), getMean(females)], 'Avg Score by Gender', ['#2563eb', '#ec4899']);

        // Division Avg
        const rx = currentData.filter(d => d.division === 'Rx').map(d => d.score);
        const scaled = currentData.filter(d => d.division === 'Scaled').map(d => d.score);
        renderChart('divisionChart', 'bar', ['Rx', 'Scaled'], [getMean(rx), getMean(scaled)], 'Avg Score by Division', ['#10b981', '#f59e0b']);
    }

    function renderChart(canvasId, type, labels, data, title, colors) {
        const ctx = document.getElementById(canvasId).getContext('2d');
        if (charts[canvasId]) {
            charts[canvasId].destroy();
        }
        charts[canvasId] = new Chart(ctx, {
            type: type,
            data: {
                labels: labels,
                datasets: [{
                    label: title,
                    data: data,
                    backgroundColor: colors,
                    borderRadius: 4
                }]
            },
            options: {
                responsive: true,
                maintainAspectRatio: false,
                plugins: {
                    legend: { display: false },
                    title: { display: true, text: title, font: {size: 14} }
                }
            }
        });
    }

    function sortTable(col) {
        if (sortCol === col) {
            sortAsc = !sortAsc;
        } else {
            sortCol = col;
            sortAsc = true;
        }
        renderTable();
    }

    function renderTable() {
        const tbody = document.getElementById('table-body');
        tbody.innerHTML = '';

        // Sort data
        const sortedData = [...currentData].sort((a, b) => {
            let valA = a[sortCol];
            let valB = b[sortCol];
            
            if (typeof valA === 'string') valA = valA.toLowerCase();
            if (typeof valB === 'string') valB = valB.toLowerCase();

            if (valA < valB) return sortAsc ? -1 : 1;
            if (valA > valB) return sortAsc ? 1 : -1;
            return 0;
        });

        // Take Top 20
        const displayData = sortedData.slice(0, 20);

        displayData.forEach(row => {
            const tr = document.createElement('tr');
            tr.innerHTML = `
                <td>${row.rank || '-'}</td>
                <td><strong>${row.athlete_name || 'Unknown'}</strong></td>
                <td>${row.gender || '-'}</td>
                <td>${row.division || '-'}</td>
                <td>${row.affiliate || '-'}</td>
                <td style="font-weight: 600;">${row.score || 0}</td>
            `;
            tbody.appendChild(tr);
        });
    }

    // Event Listeners
    document.getElementById('filter-gender').addEventListener('change', applyFilters);
    document.getElementById('filter-division').addEventListener('change', applyFilters);

    // Initialize
    loadData();

</script>
